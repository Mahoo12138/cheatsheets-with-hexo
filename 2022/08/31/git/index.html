Git 常用操作手册 2.9.1

#get started

#Get Start

user config

git config --global user.name "xxx"
git config --global user.email "xxx"
Set default editor
git config --global core.editor "vim"

link remote repo

# 生成 ssh 密钥
ssh-keygen -t rsa -C "xxx@xxx.com"
# 检查连接
ssh -T git@xxx.com

#Branch

create new branch

git checkout -b feature/a
git checkout -b feature/b origin/master

detele local branch

git branch -d &lt;branch-name>

delete remote branch

git push origin --delete &lt;branch-name>

rename local branch

git branch -m oldName newName

rename remote branch

# 重命名远程分支对应的本地分支
git branch -m oldName newName
# 删除远程分支
git push --delete origin oldName
# 上传新命名的本地分支
git push origin newName
# 把修改的本地分支与远程分支关联
git branch --set-upstream-to origin/newName

unset upstream

git branch --unset-upstream

#Sync

sync the upstream

# 添加一个远程分支作为上游仓库
git remote add upstream &lt;upstream-url>
# 提交或暂存修改 
git commit / git stash
# 拉取上游分支改动
git fetch upstream
# 切换到主分支
git checkout master
# 将上游分支改动合并到本地主分支
git merge upstream/master

replace remote repo

# 删除当前远程仓库关联
git remote rm origin
# 添加新的远程仓库
git remote add origin [repo-url]
# 验证更改
git remote -v

#Tag

create new tag

 git tag v1.0.0

rename a tag

 # 基于旧标签创建新标签
 git tag &lt;new-tag-name> &lt;old-tag-name>
 # 删除旧的标签
 git tag -d &lt;old-tag-name>
 # 同步到远程分支
 git push origin :refs/tags/&lt;old-tag-name>

sync tag to remote

 git push origin --tags
