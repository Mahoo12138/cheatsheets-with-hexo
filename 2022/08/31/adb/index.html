安装 App

# 覆盖安装
adb install &lt;apk file> -r
# 安装到sdcard
adb install &lt;apk file> -s
# 允许降级覆盖安装
adb install &lt;apk file> -d

卸载 App

# -k 参数为是否保留数据和缓存
adb unstall -k &lt;package-name>

清除应用数据与缓存

adb shell pm clear &lt;package-name>

远程连接 ADB

# 同一局域网下，连接数据线
adb tcpip 5555
# 拔除数据线
adb connect &lt;device_ip_address>

查看IP地址

# interface name for example: wlan0
adb shell ifconfig &lt;interface name>
